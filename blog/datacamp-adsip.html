import numpy as np

sales = [120, 150, 90, 200, 175]
sales = np.array(sales)

def normalise(values):
    return (values - np.mean(values)) / np.std(values)

z_scores = normalise(sales)

print("Z-scores:", np.round(z_scores, 2))

import pandas as pd
import matplotlib.pyplot as plt

# Dictionary ‚Üí DataFrame
data = {
    "day": ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"],
    "steps": [8200, 7600, 10400, 9800, 12000, 15000, 13200]
}

df = pd.DataFrame(data)

# Logic & filtering
active_days = df[df["steps"] > 10000]

# Loop + list comprehension
labels = []
for steps in df["steps"]:
    if steps >= 12000:
        labels.append("High")
    elif steps >= 9000:
        labels.append("Medium")
    else:
        labels.append("Low")

df["activity_level"] = labels

# Matplotlib: multiple plots + customization
fig, axes = plt.subplots(1, 2, figsize=(10, 4))

# Bar plot
axes[0].bar(df["day"], df["steps"], color="steelblue")
axes[0].axhline(10000, linestyle="--", color="red", alpha=0.7)
axes[0].set_title("Daily Step Count")
axes[0].set_ylabel("Steps")

# Scatter plot
axes[1].scatter(df["day"], df["steps"], c=df["steps"], cmap="viridis")
axes[1].set_title("Steps by Day (Colored by Value)")
axes[1].set_ylabel("Steps")

plt.tight_layout()
plt.show()

Data Manipulation with pandas

Module 3

Transforming DataFrames, Aggregating DataFrames, Slicing and Indexing DataFrames, and Creating and Visualizing DataFrames:
- Transforming DataFrames: adding, modifying, removing columns
- Aggregating DataFrames: groupby, pivot tables, summary statistics
- Slicing and Indexing DataFrames: loc, iloc, boolean indexing
- Creating and Visualizing DataFrames: creating DataFrames from scratch, basic plotting with pandas 

import pandas as pd
import matplotlib.pyplot as plt

# Create DataFrame from scratch
data = {
    "player": ["A", "B", "C", "A", "B", "C"],
    "week": [1, 1, 1, 2, 2, 2],
    "score": [120, 95, 110, 135, 105, 125]
}

df = pd.DataFrame(data)

# Transforming DataFrames: add / modify columns
df["score_change"] = df.groupby("player")["score"].diff()
df["high_score"] = df["score"] >= 120

# Slicing and indexing
week_two = df.loc[df["week"] == 2, ["player", "score"]]
first_three_rows = df.iloc[:3]

# Aggregation: groupby + summary stats
summary = (
    df.groupby("player")["score"]
      .agg(mean_score="mean", max_score="max")
)

# Pivot table
pivot = df.pivot_table(
    values="score",
    index="week",
    columns="player",
    aggfunc="mean"
)

# Basic pandas plotting
summary["mean_score"].plot(
    kind="bar",
    title="Average Score by Player",
    ylabel="Score",
    xlabel="Player",
    color="steelblue"
)

plt.tight_layout()
plt.show()

Module 4

Data merging basics, merging tables with different join types, advances merging and concatenating, merging ordered time-series data:
- Data merging basics: merge function, common columns, suffixes
- Merging tables with different join types: inner, outer, left, right joins
- Advanced merging and concatenating: multi-index merges, concatenation along rows/columns
- Merging ordered time-series data: asof merges, time-based joins 

import pandas as pd

# Create DataFrames
df1 = pd.DataFrame({
    "key": ["A", "B", "C"],
    "value1": [1, 2, 3]
})

df2 = pd.DataFrame({
    "key": ["B", "C", "D"],
    "value2": [4, 5, 6]
})

# Basic merging
merged = pd.merge(df1, df2, on="key", how="inner")

# Advanced merging
df1.set_index("key", inplace=True)
df2.set_index("key", inplace=True)
merged = df1.join(df2, how="outer", lsuffix="_left", rsuffix="_right")

# Concatenation
concatenated = pd.concat([df1, df2], axis=0)

# Time-based merging
time_indexed = pd.DataFrame({
    "date": pd.date_range("2023-01-01", periods=3),
    "value": [1, 2, 3]
}).set_index("date")

asof_merged = pd.merge_asof(time_indexed, time_indexed, on="date", direction="backward")
